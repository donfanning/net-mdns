        def to_s
          s = "#{name.to_s} (#{ttl}) "
          s << '!' if absolute?
          s << '-' if ttl == 0
          s << " #{DNS.rrname(data)}"

          case data
          when IN::A
            s << " #{data.address.to_s}"
          when IN::PTR
            s << " #{data.name}"
          when IN::SRV
            s << " #{data.target}:#{data.port}"
          when IN::TXT
            s << " #{data.strings.first.inspect}#{data.strings.length > 1 ? ', ...' : ''}"
          when IN::HINFO
            s << " os=#{data.os}, cpu=#{data.cpu}"
          else
            s << data.inspect
          end
          s
        end
